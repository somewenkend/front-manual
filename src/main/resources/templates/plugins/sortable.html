    var sortArray = [];
    Sortable.create($('#items')[0], { // 参数是要排序的容器dom对象
        dataIdAttr: "data-id", // 配置排序项关联的数据
        delay: 0,//延迟
        animation: 150,//动画
        onStart: function (evt) { // 拖拽开始回调方法
        },
        onEnd: function (evt) { // 拖拽结束回调方法
        },
        onSort: function (evt) { // 列表排完序后调用
            sortArray = this.toArray(); // 处理后的dataIdAttr新数组
            alert("排序后的数组：" + sortArray);
        }
    });
